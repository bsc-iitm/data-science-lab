def min_sec(s):
  sec = 0
  a = s.split() # ["2h", "30m"]
  for i in range(len(a)):
    if "h" in a[i]:
     sec += int(a[i][:-1])*60*60
    if "m" in a[i]:
      sec += int(a[i][:-1])*60
  return sec
data["new_duration"] = data['Duration'].apply(min_sec)

##Transforming the values in the â€˜Departure Timeâ€™ and â€˜Arrival Timeâ€™ columns to represent the hour component. For instance, if an entry is 10:05, the corresponding value should be 10. ##Then converting the time into four categories as follows: ##5 <= hour < 12 = Morning ##12 <= hour < 17 = Afternoon ##17 <= hour < 20 = Evening ##20 <= hour < 5 = Night

def hr(time):
  time = time.split(':')
  hour = int(time[0])
  if 5 <= hour and hour < 12:
    return 'Morning'
  if 12 <= hour and hour < 17:
    return 'Afternoon'
  if 17 <= hour and hour < 20:
    return 'Evening'
  if 20 <= hour or hour < 5:
    return 'Night'
data['Arrival Time'] = data['Arrival Time'].apply(hr)
